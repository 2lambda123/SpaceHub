278  stl_ranges.clear();
279  stl_ranges.reserve(impl_number() * 6 + 3);

367  template<typename Scalar, ReguType Type>
368  template<typename Particles>
369  Scalar Regularization<Scalar, Type>::eval_pos_phy_time(const Particles &particles, Scalar step_size) const {
370  if constexpr (Type == ReguType::LogH) {
371  return step_size / (bindE_ + calc::calc_kinetic_energy(particles));
372  } else if constexpr (Type == ReguType::TTL) {
373  return step_size / omega_;
374  } else if constexpr (Type == ReguType::None) {
375  return step_size;
376  } else {
377  spacehub_abort("Undefined regularization type!");
378  }
379  }

383  Scalar Regularization<Scalar, Type>::eval_vel_phy_time(const Particles &particles, Scalar step_size) const {
384  if constexpr (Type == ReguType::LogH) {
385  return step_size / -calc::calc_potential_energy(particles);

388  } else if constexpr (Type == ReguType::None) {
