170  template<typename TypeSystem>
171  template<typename STL>
172  PointParticles<TypeSystem>::PointParticles(Scalar t, const STL &particle_set) {
173  static_assert(is_container_v<STL>, "Only STL-like container can be used");
174  SPACEHUB_PARTICLE_TYPE_CHECK(STL, Particle);
175 
176  size_t input_num = particle_set.size();
177  this->reserve(input_num);
178  size_t id = 0;
179  for (auto &p : particle_set) {
180  pos_.emplace_back(p.pos);
181  vel_.emplace_back(p.vel);
182  mass_.emplace_back(p.mass);
183  idn_.emplace_back(id++);
184  }
185  time_ = t;
186  active_num_ = input_num;
187  }
