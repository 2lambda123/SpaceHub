158  stl_ranges.reserve(coords.size() * 3 + stl_ranges.size());
159  std::copy(coords.x.begin(), coords.x.end(), std::back_insert_iterator(stl_ranges));

164 template <typename STLIterator, typename T>
165 void load_to_coords(STLIterator iter_start, STLIterator iter_end, Coords<T> &coords) {
166  size_t len = (iter_end - iter_start)/3;
167  coords.resize(len);
168  auto iter = iter_start;
169  for (auto &xx : coords.x) {
170  xx = *iter++;
171  }
172  for (auto &yy : coords.y) {
173  yy = *iter++;
174  }
175  for (auto &zz : coords.z) {
176  zz = *iter++;
177  }
178 }

180 template <typename T>
181 auto distance(Coords<T> const &c, size_t i, size_t j) {
182  auto dx = c.x[i] - c.x[j];
183  auto dy = c.y[i] - c.y[j];
184  auto dz = c.z[i] - c.z[j];
185  return sqrt(dx * dx + dy * dy + dz * dz);
186 }
