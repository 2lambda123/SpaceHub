101  if constexpr (std::tuple_size_v<ArrayLike> == 1) {//Only proportion part is provided

104  step_limiter(order, safe_guard1_ * pow(safe_guard2_ / std::get<0>(errors), expon_[order]));
105  } else {

112  pow(std::get<1>(errors), Ki_ * expon_[order]));
