54  integrator_.evaluateSystemAt(particles, iterH, Integrator::finalPoint);

96  bool isConvergent(const RadauTab &BTab, const RadauTab &newBTab, const VectorArray &acc) {
97  size_t size = acc.size();
98 
99  Scalar diff = 0;
100  Scalar scale = 0;
101  for (size_t i = 0; i < size; ++i) {
102  diff = space::max(diff, (BTab[i][6] - newBTab[i][6]).abs().max_component());
103  scale = space::max(scale, acc[i].abs().max_component());
104  }
105 
106  Scalar convergence = diff / scale;
