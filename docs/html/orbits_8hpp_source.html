247  void coord_to_oribt_args(Scalar m1, Scalar m2, const Vector &dr, const Vector &dv, OrbitArgs<Scalar> &args) {
248  Vector L = cross(dr, dv);
249  Vector N = cross(Vector(0, 0, 1.0), L);
250  Scalar r = norm(dr);
251  Scalar n = norm(N);
252  Scalar l = norm(L);
253  Scalar rv = dot(dr, dv);
254  Scalar u = (m1 + m2) * consts::G;
255  Vector E = (dr * (norm2(dv) - u * re_norm(dr)) - dv * rv) / u;
256 
257  args.m1 = m1;
258  args.m2 = m2;
259  args.e = norm(E);
260  args.orbit_type = classify_orbit(args.e);
261 
262  if (args.orbit_type == OrbitType::Parabola) {
263  Scalar a = -u * r / (r * norm2(dv) - 2.0 * u);
264  args.p = semi_latus_rectum(a, args.e);
265  } else {
266  args.p = l * l / u;
267  }
268 
269  args.i = acos(L.z / l);
270 
271  if (args.e != 0) {
272  args.nu = space::sign(rv) * myacos(dot(E / args.e, dr / r));
273 
274  if (n != 0) {
275  args.Omega = space::sign(N.y) * myacos(N.x / n);
276  args.omega = space::sign(E.z) * myacos(dot(E / args.e, N / n));
277  } else {
278  args.omega = -space::sign(E.y) * myacos(-E.x / args.e);
279  args.Omega = args.omega;
280  }
281  } else {

283  args.Omega = space::sign(N.y) * myacos(N.x / n);
284  args.omega = 0;
285  Vector peri = cross(L, N);
286  args.nu = -space::sign(dot(N, dr)) * myacos(dot(peri / norm(peri), dr / r));
287  } else {
288  args.Omega = args.omega = 0;
289  args.nu = space::sign(dr.y) * acos(dot(Vector(1.0, 0, 0), dr / r));
