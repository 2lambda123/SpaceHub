43  template<typename Vector, typename Scalar>
44  void euler_rotate(Vector &v, const Scalar phi, const Scalar theta, const Scalar psi) {
45  Scalar sin_phi = sin(phi);
46  Scalar cos_phi = cos(phi);
47  Scalar sin_psi = sin(psi);
48  Scalar cos_psi = cos(psi);
49  Scalar sin_theta = sin(theta);
50  Scalar cos_theta = cos(theta);
51 
52  Scalar x = v.x * (cos_phi * cos_psi - sin_phi * cos_theta * sin_psi) -
53  v.y * (cos_phi * sin_psi + sin_phi * cos_theta * cos_psi) + v.z * (sin_phi * sin_theta);
54 
55  Scalar y = v.x * (sin_phi * cos_psi + cos_phi * cos_theta * sin_psi) -
56  v.y * (sin_phi * sin_psi - cos_phi * cos_theta * cos_psi) - v.z * (cos_phi * sin_theta);
57 
58  Scalar z = v.x * sin_theta * sin_psi + v.y * sin_theta * cos_psi + v.z * cos_theta;
59 
60  v.x = x, v.y = y, v.z = z;
61  }

63  template<typename Scalar>
64  Scalar calc_true_anomaly(Scalar eccentric_anomaly, Scalar e) {
65  if (0 <= e && e < 1)
66  return 2 * atan2(sqrt(1 + e) * sin(eccentric_anomaly * 0.5), sqrt(1 - e) * cos(0.5 * eccentric_anomaly));
67  else if (e > 1)
68  return 2 * atan2(sqrt(1 + e) * sinh(eccentric_anomaly * 0.5), sqrt(e - 1) * cosh(0.5 * eccentric_anomaly));
69  else if (iseq(e, 1.0))
70  return 2 * atan(0.5 * eccentric_anomaly);
71  else {
72  spacehub_abort("Eccentricity cannot be negative, Nan or inf!");
73  }
74  }

90  template<typename Scalar>
91  Scalar calc_eccentric_anomaly(Scalar mean_anomaly, Scalar e) {
92  if (fabs(mean_anomaly) <= space::epsilon<Scalar>::value) return 0;
93 
94  if (0 <= e && e < 1)
95  return space::root_bisection(
96  [&](Scalar x) -> Scalar { return (x - e * sin(x) - mean_anomaly) / (1 - e * cos(x)); },
97  -space::consts::pi, space::consts::pi); // find this function in ownMath.h
98  else if (e > 1)
99  return space::root_bisection(
100  [&](Scalar x) -> Scalar { return (e * sinh(x) - x - mean_anomaly) / (e * cosh(x) - 1); },
101  -space::consts::pi, space::consts::pi);
102  else if (fabs(e - 1) < space::epsilon<Scalar>::value)
103  return space::root_bisection([&](Scalar x) -> Scalar { return (x + x * x * x / 3 - mean_anomaly) / (1 + x * x); },
104  -space::consts::pi, space::consts::pi);
105  else {
106  spacehub_abort("Eccentricity cannot be negative, Nan or inf!");
107  }

272  template<typename Vector, typename Scalar>
273  void coord_to_oribt_args(Scalar m1, Scalar m2, const Vector &dr, const Vector &dv, OrbitArgs<Scalar> &args) {
274  Vector L = cross(dr, dv);
275  Vector N = cross(Vector(0, 0, 1.0), L);
276  Scalar r = norm(dr);
277  Scalar n = norm(N);
278  Scalar l = norm(L);
279  Scalar rv = dot(dr, dv);
280  Scalar u = (m1 + m2) * consts::G;
281  Vector E = (dr * (norm2(dv) - u * re_norm(dr)) - dv * rv) / u;
282 
283  args.m1 = m1;
284  args.m2 = m2;
285  args.e = norm(E);
286  args.orbit_type = classify_orbit(args.e);
287 
288  if (args.orbit_type == OrbitType::Parabola) {
289  Scalar a = -u * r / (r * norm2(dv) - 2.0 * u);
290  args.p = semi_latus_rectum(a, args.e);
291  } else {
292  args.p = l * l / u;
293  }
294 
295  args.i = acos(L.z / l);
296 
297  if (args.e != 0) {
298  args.nu = space::sign(rv) * myacos(dot(E / args.e, dr / r));
299 
300  if (n != 0) {
301  args.Omega = space::sign(N.y) * myacos(N.x / n);
302  args.omega = space::sign(E.z) * myacos(dot(E / args.e, N / n));
303  } else {
304  args.omega = -space::sign(E.y) * myacos(-E.x / args.e);
305  args.Omega = args.omega;
306  }
307  } else {
308  if (n != 0) {
309  args.Omega = space::sign(N.y) * myacos(N.x / n);
310  args.omega = 0;
311  Vector peri = cross(L, N);
312  args.nu = -space::sign(dot(N, dr)) * myacos(dot(peri / norm(peri), dr / r));
313  } else {
314  args.Omega = args.omega = 0;
315  args.nu = space::sign(dr.y) * acos(dot(Vector(1.0, 0, 0), dr / r));
316  }
317  }
318  }

320  template<typename Vector, typename Scalar>
321  void oribt_args_to_coord(OrbitArgs<Scalar> const &args, Vector &pos, Vector &vel) {
322  Scalar u = (args.m1 + args.m2) * consts::G;
323 
324  Scalar sin_nu = sin(args.nu);
325  Scalar cos_nu = cos(args.nu);
326 
327  Scalar r = args.p / (1 + args.e * cos_nu);
328  Scalar v = sqrt(u / args.p);
329 
330  pos = r * Vector(cos_nu, sin_nu, 0);
331  vel = v * Vector(-sin_nu, args.e + cos_nu, 0);
332 
333  orbit::euler_rotate(pos, args.Omega, args.i, args.omega + consts::pi);
334  orbit::euler_rotate(vel, args.Omega, args.i, args.omega + consts::pi);

337  template<typename Particle, typename... Args>
338  void move_to_com_coord(Particle &ptc, Args &... ptcs) {
339  if constexpr (sizeof...(Args) != 0) {
340  static_assert(calc::all(std::is_same_v<Args, Particle>...), "Wrong particles type!");
341  auto tot_mass = (ptcs.mass + ... + ptc.mass);
342  auto cm_pos = ((ptcs.mass * ptcs.pos) + ... + (ptc.mass * ptc.pos)) / tot_mass;
343  auto cm_vel = ((ptcs.mass * ptcs.vel) + ... + (ptc.mass * ptc.vel)) / tot_mass;
344  ((ptc.pos -= cm_pos), ..., (ptcs.pos -= cm_pos));
345  ((ptc.vel -= cm_vel), ..., (ptcs.vel -= cm_vel));
346  } else if constexpr (is_container_v<Particle>) {
347  using sParticle = typename Particle::value_type;
348  using Scalar = typename sParticle::Scalar;
349  using Vector = typename sParticle::Vector;
350 
351  Scalar tot_mass{0};
352  Vector cm_pos{0};
353  Vector cm_vel{0};
354 
355  for (auto const &p : ptc) {
356  tot_mass += p.mass;
357  cm_pos += p.mass * p.pos;
358  cm_vel += p.mass * p.vel;
359  }
360 
361  cm_pos /= tot_mass;
362  cm_vel /= tot_mass;
363 
364  for (auto &p : ptc) {
365  p.pos -= cm_pos;
366  p.vel -= cm_vel;
367  }
368  }
369  }

436  template<typename Vector, typename Scalar>
437  auto calc_a_e(Scalar u, Vector const &dr, Vector const &dv) {
438  Scalar r = norm(dr);
439  Scalar v = norm(dv);
440  Scalar v2 = v * v;
441  Scalar vr = dot(dr, dv);
442  Scalar vdfs = v2 - u / r;
443  Scalar a = -u / (v2 - 2 * u / r);
444  Scalar e = norm((dr * vdfs - dv * vr) / u);
445  return std::make_tuple(a, e);
