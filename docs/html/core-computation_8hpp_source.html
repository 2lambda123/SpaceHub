219  template<typename Particles>
220  auto calc_potential_energy(Particles const &particles) {
221 
222  typename Particles::Scalar potential_eng{0};
223  size_t const size = particles.number();
224  auto const &m = particles.mass();
225  auto const &v = particles.vel();
226  auto const &p = particles.pos();
227 
228  if constexpr (HAS_METHOD(Particles, chain_pos) && HAS_METHOD(Particles, index)) {
229  auto const &ch_px = particles.chain_pos().x;
230  auto const &ch_py = particles.chain_pos().y;
231  auto const &ch_pz = particles.chain_pos().z;
232  auto const &idx = particles.index();
233 
234  for (size_t i = 0; i < size - 1; ++i) {
235  potential_eng -=
236  m[idx[i]] * m[idx[i + 1]] / sqrt(ch_px[i] * ch_px[i] + ch_py[i] * ch_py[i] + ch_pz[i] * ch_pz[i]);
237  }
238 
239  for (size_t i = 0; i < size - 2; ++i) {
240  auto dx = ch_px[i] + ch_px[i + 1];
241  auto dy = ch_py[i] + ch_py[i + 1];
242  auto dz = ch_pz[i] + ch_pz[i + 1];
243  potential_eng -= m[idx[i]] * m[idx[i + 2]] / sqrt(dx * dx + dy * dy + dz * dz);
244  }
245 
246  for (size_t i = 0; i < size; ++i) {
247  for (size_t j = i + 3; j < size; ++j) {
248  auto dx = p.x[idx[j]] - p.x[idx[i]];
249  auto dy = p.y[idx[j]] - p.y[idx[i]];
250  auto dz = p.z[idx[j]] - p.z[idx[i]];
251  potential_eng -= m[idx[i]] * m[idx[j]] / sqrt(dx * dx + dy * dy + dz * dz);
252  }
253  }
254  } else {
255  for (size_t i = 0; i < size; ++i)
256  for (size_t j = i + 1; j < size; ++j) {
257  auto dx = p.x[i] - p.x[j];
258  auto dy = p.y[i] - p.y[j];
259  auto dz = p.z[i] - p.z[j];
260  potential_eng -= m[i] * m[j] / sqrt(dx * dx + dy * dy + dz * dz);
261  }
262  }
263  return potential_eng;
264  }

271 /** @brief Calculate the minimal fall free time of two particles
272  *
273  * @param mass mass array of particle.
274  * @param position position array of particle.
275  * @return The minimal fall free time of the two particles
276  */
277  template<typename ScalarArray, typename Coord>
278  inline auto calc_fall_free_time(ScalarArray const &mass, Coord const &position) {
279  using Scalar = typename ScalarArray::value_type;
280  size_t const size = mass.size();
281  Scalar min_fall_free = max_value<Scalar>::value;
282 
283  for (size_t i = 0; i < size; i++) {
284  for (size_t j = i + 1; j < size; j++) {
285  Scalar dx = position.x[i] - position.x[j];
286  Scalar dy = position.y[i] - position.y[j];
287  Scalar dz = position.z[i] - position.z[j];
288  Scalar r = sqrt(dx * dx + dy * dy + dz * dz);
289  Scalar fall_free = pow(r, 1.5) / sqrt(mass[i] + mass[j]);
290 
291  if (fall_free < min_fall_free) min_fall_free = fall_free;
292  }
293  }
294  return min_fall_free * space::consts::pi * 0.5 / sqrt(2 * space::consts::G);
295  }
