191  template<typename Particles, typename Interactions>
192  void ChainSystem<Particles, Interactions>::impl_kick(Scalar step_size) {
193  if constexpr (Interactions::ext_vel_dep) {
194  Scalar half_step = 0.5 * step_size;
195  eval_vel_indep_acc();
196  kick_pseu_vel(half_step);
197  kick_real_vel(step_size);
198  kick_pseu_vel(half_step);
199  } else {
200  interactions_.eval_acc(*this, accels_.acc());
201  impl_advance_vel(accels_.acc(), step_size);
202  }

214  void ChainSystem<Particles, Interactions>::impl_post_iter_process() {
215  Chain::calc_chain_index(ptcl_.pos(), new_index_);
216  if (new_index_ != index_) {
217  Chain::update_chain(chain_pos_, index_, new_index_);
218  Chain::calc_cartesian(ptcl_.mass(), chain_pos_, ptcl_.pos(), new_index_);
219  Chain::update_chain(chain_vel_, index_, new_index_);
220  Chain::calc_cartesian(ptcl_.mass(), chain_vel_, ptcl_.vel(), new_index_);
221  index_ = new_index_;
