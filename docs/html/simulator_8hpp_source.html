38 /**
39  * Run arguments that is used to set all arguments needed by Simulator.
40  *
41  * The pre_operations(), post_operations(), stop_operations() and check_stops() will be called in a simulation
42  * in this way.
43  *
44  * @code{.cpp}
45  * ...
46  * auto sys; // particle system that is going to be evolved.
47  * for( ;check_stops(sys);){
48  * pre_operations(sys);
49  * ...//step integration.
50  * post_operations(sys);
51  * }
52  * stop_operations(sys);
53  * ...
54  * @endcode
55  *
56  * @tparam ParticleSys Any implementation of concept ParticleSystem.
57  */

358  template<typename ParticleSys, typename OdeIterator>
359  void Simulator<ParticleSys, OdeIterator>::run(RunArgs const &run_args) {
360  if (!run_args.is_stop_condition_set() && !run_args.is_end_time_set()) {
361  space::spacehub_abort("Use 'add_stop_condition' to set stop condition.");
362  }
363 
364  step_size_ = run_args.step_size;
365 
366  if (step_size_ == 0.0) {
367  step_size_ = 0.01 * calc::calc_step_scale(particles_);
368  }
369 
370  Scalar end_time = space::unit::hubble_t;
371 
372  if (run_args.is_end_time_set()) {
373  end_time = run_args.end_time;
374  }
375 
376  if (particles_.time() >= end_time) {
377  space::print(std::cout, "Warning: The stop time is '<=' to the start time!");
378  }
379 
380  if constexpr (HAS_CRTP_IMPLEMENTATION(OdeIterator, set_atol, Scalar)) {
381  iterator_.set_atol(run_args.atol);
382  }
383 
384  if constexpr (HAS_CRTP_IMPLEMENTATION(OdeIterator, set_rtol, Scalar)) {
385  iterator_.set_rtol(run_args.rtol);
386  }
387 
388  for (; particles_.time() < end_time && !run_args.check_stops(particles_);) {
389 
390  run_args.pre_operations(particles_);
391  advance_one_step();
392  run_args.post_operations(particles_);
393  }
394  run_args.stop_operations(particles_);
395  }
396 
397  template<typename ParticleSys, typename OdeIterator>
398  inline void Simulator<ParticleSys, OdeIterator>::advance_one_step() {
399  particles_.pre_iter_process();
400  step_size_ = iterator_.iterate(particles_, step_size_);
401  particles_.post_iter_process();
402  }
