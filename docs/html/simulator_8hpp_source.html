283  void Simulator<ParticSys, OdeIterator>::run(RunArgs const &arg) {
284  if (!arg.is_stop_condition_set() && !arg.is_end_time_set()) {
285  space::spacehub_abort("Use 'add_stop_condition' to set stop condition.");
286  }
287 
288  step_size_ = arg.step_size;
289 
290  if (step_size_ == 0.0) {
291  step_size_ = 0.01 * calc::calc_step_scale(particles_);
292  }
293 
294  Scalar end_time = space::unit::hubble_t;
295 
296  if (arg.is_end_time_set()) {
297  end_time = arg.end_time;
298  }
299 
300  if (particles_.time() >= end_time) {
301  space::print(std::cout, "Warning: The stop time is '<=' to the start time!");
302  }
303 
304  if constexpr (HAS_CRTP_IMPLEMENTATION(OdeIterator, set_atol, Scalar)) {
305  iterator_.set_atol(arg.atol);
306  }
307 
308  if constexpr (HAS_CRTP_IMPLEMENTATION(OdeIterator, set_rtol, Scalar)) {
309  iterator_.set_rtol(arg.rtol);
310  }
311 
312  for (; particles_.time() < end_time && !arg.check_stops(particles_);) {
313 
314  arg.pre_operations(particles_);
315  advance_one_step();
316  arg.post_operations(particles_);
317  }
318  arg.stop_operations(particles_);
319  }
