256  template<typename Particles, typename Interactions, ReguType RegType>
257  void ARchainSystem<Particles, Interactions, RegType>::impl_post_iter_process() {
258  Chain::calc_chain_index(ptcl_.pos(), new_index_);
259  if (new_index_ != index_) {
260  Chain::update_chain(chain_pos_, index_, new_index_);
261  Chain::calc_cartesian(ptcl_.mass(), chain_pos_, ptcl_.pos(), new_index_);
262  Chain::update_chain(chain_vel_, index_, new_index_);
263  Chain::calc_cartesian(ptcl_.mass(), chain_vel_, ptcl_.vel(), new_index_);
264  index_ = new_index_;
265  }
