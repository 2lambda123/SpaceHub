49  template<typename Lambda>
50  void multi_threads_loop(size_t total_len, size_t thread_num, Lambda &&task) {
51  auto len_pth = total_len / thread_num;
52  std::vector<std::thread> threads;
53  threads.reserve(thread_num);
54  for (size_t th_id = 0; th_id < thread_num; ++th_id) {
55  size_t begin = th_id * len_pth;
56  size_t end = begin + len_pth;
57  if (end + len_pth > total_len)
58  end = total_len;
59  threads.emplace_back(std::thread(std::forward<Lambda>(task), begin, end));
60  }
61 
62  for (auto &thread : threads) {
63  if (thread.joinable())
64  thread.join();
65  }
66  }

68  template<typename Callable, typename ...Args>
69  void multi_thread_run(size_t thread_num, Callable &&job, Args &&...args) {
70  std::vector<std::thread> threads;
71  threads.reserve(thread_num);
72  for (size_t i = 0; i < thread_num; ++i) {
73  threads.emplace_back(std::thread(std::forward<Callable>(job), std::forward<Args>(args)...));
74  }
75 
76  for (auto &th : threads) {
77  if (th.joinable())
78  th.join();
79  }
