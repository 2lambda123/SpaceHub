59  using Vector = typename Coord::Vector;

61 
62  Coord new_chain;
63  new_chain.reserve(size);
64 
65  auto get_idx = [&](auto var) -> auto { return std::find(idx.begin(), idx.end(), var) - idx.begin(); };

67  for (size_t i = 0; i < size - 1; ++i) {
68  auto first = get_idx(new_idx[i]);

120  static bool try_add_to_chain(std::list<size_t> &list, size_t &head, size_t &tail, Node &n) {
121  if (head == n.i) {
122  return try_insert(list, head, n, n.j, [&](size_t idx) { list.emplace_front(idx); });
123  } else if (head == n.j) {
124  return try_insert(list, head, n, n.i, [&](size_t idx) { list.emplace_front(idx); });
125  } else if (tail == n.i) {
126  return try_insert(list, tail, n, n.j, [&](size_t idx) { list.emplace_back(idx); });
127  } else if (tail == n.j) {
128  return try_insert(list, tail, n, n.i, [&](size_t idx) { list.emplace_back(idx); });
129  }
130  return false;
131  }

133  template<typename Coord, typename Container>
134  static void create_distances_array(Coord const &pos, Container &vec) {
135  size_t num = pos.size();
136  vec.reserve(num * (num - 1));
137  for (size_t i = 0; i < num; ++i) {
138  for (size_t j = i + 1; j < num; ++j) {
139  auto dx = pos.x[j] - pos.x[i];
140  auto dy = pos.y[j] - pos.y[i];
141  auto dz = pos.z[j] - pos.z[i];
142  vec.emplace_back(Node{dx * dx + dy * dy + dz * dz, i, j, true});
143  }
144  }
145  }

148  static void create_index_from_dist_array(std::vector<Node> &dist, IdxArray &idx, size_t num) {
149  auto head = dist[0].i;
150  auto tail = dist[0].j;
151  dist[0].avail = false;
152 
153  std::list<size_t> idx_list{head, tail};
154 
155  size_t chained_num = 1;
156 
157  size_t dist_size = dist.size();
158  for (size_t k = 1; k < dist_size; ++k) {
159  if (dist[k].avail) {
160  if (try_add_to_chain(idx_list, head, tail, dist[k])) {
161  chained_num++;
162  if (chained_num == num) {
163  break;
164  } else {
165  k = 1;
166  }
167  }
168  }
169  }

172  for (auto &l : idx_list) {
173  idx.emplace_back(l);

177  template<typename Coord>
178  static auto get_new_node(Coord const &chain, size_t head, size_t tail) -> typename Coord::Vector {
179  using Scalar = typename Coord::Scalar;
180  using Vector = typename Coord::Vector;
181 
182  Scalar sign{1};
183 
184  if (head > tail) {
185  std::swap(head, tail);
186  sign = -1;
187  }
188 
189  auto connect = [](auto const &array, auto first, auto last) -> auto {
190  auto new_d = array[first];
191  for (size_t j = first + 1; j < last; ++j) {
192  new_d += array[j];
193  }
194 
195  return new_d;
196  };
197 
198  return Vector(sign * connect(chain.x, head, tail), sign * connect(chain.y, head, tail),
199  sign * connect(chain.z, head, tail));
200  }

202  template<typename Array, typename IdxArray>
203  static void to_chain(Array const &cartesian, Array &chain, IdxArray const &index) {
204  const size_t size = cartesian.size();
205  if constexpr (!auto_centre_of_mass_move_) {
206  chain[size - 1] = 0;
