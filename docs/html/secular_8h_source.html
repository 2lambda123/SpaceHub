33 double t_k_quad(double m_in, double m_out, double a_in, double a_out, double c_out) {
34  double rc = a_out * c_out;
35  return 1.0 / sqrt(G) * sqrt(m_in) / m_out * rc * rc * rc * pow(a_in, -1.5);
36 }

84  double coef = 0.75 / t_k_quad(m1 + m2, m3, calc_a_in(L_in, c_in), calc_a_out(L_out, c_out), c_out);

86  Vec3d dL1dt = coef * L_in * ((c_in * c_in * dn1n2) * cn1n2 - (5 * de1n2) * ce1n2);

88  Vec3d dL2dt = -dL1dt;

92  Vec3d de2dt = coef * (L_in / L_out) / c_out *
93  ((c_in * c_in * dn1n2) * ce2n1 + (5 * de1n2) * ce1e2 -
94  (0.5 - 3 * de1e1 + 12.5 * de1n2 * de1n2 - 2.5 * c_in * c_in * dn1n2 * dn1n2) * cn2e2);
