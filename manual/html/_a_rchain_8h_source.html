346 void ARchain<Interaction, EvolvedData, Regularitor>::updateVelIndepAcc()
347 {
348  Vector dr(0.0, 0.0, 0.0);
349  Scalar inv_r = 1;
350  Scalar inv_r3 = 1;
351  memset(&(velIndepAcc[0]), 0, sizeof(Vector)*size());
352 
353  for(size_t i = 0 ; i < size() - 1; ++i)
354  {
355  dr = chain.pos[i];
356  inv_r = dr.reNorm();
357  inv_r3 = (inv_r * inv_r * inv_r);
358  velIndepAcc[chainIndex[i]] += dr * (inv_r3 * this->mass[chainIndex[i + 1]]);
359  velIndepAcc[chainIndex[i + 1]] -= dr * (inv_r3 * this->mass[chainIndex[i]]);
360  }
361 
362  for(size_t i = 0 ; i < size() - 2; ++i)
363  {
364  dr = chain.pos[i] + chain.pos[i + 1];
365  inv_r = dr.reNorm();
366  inv_r3 = (inv_r * inv_r * inv_r);
367  velIndepAcc[chainIndex[i]] += dr * (inv_r3 * this->mass[chainIndex[i + 2]]);
368  velIndepAcc[chainIndex[i + 2]] -= dr * (inv_r3 * this->mass[chainIndex[i]]);
369  }

376  inv_r = dr.reNorm();
377  inv_r3 = (inv_r * inv_r * inv_r);

393 template<typename Interaction, typename EvolvedData, typename Regularitor>
394 void ARchain<Interaction, EvolvedData, Regularitor>::updateAccWith(VectorArray& velocity, VectorArray& chainVel)
395 {
396  memset(&(velDepAcc[0]), 0, sizeof(Vector)*size());
397  Vector dr;
398  Vector dv;
399 
400  for(int i = 1 ; i < size(); ++i)//update chain data(near pair)
401  {
402  dr = chain.pos[i - 1];
403  dv = chainVel[i - 1];
404  velDepForce(this->mass[chainIndex[i]], this->mass[chainIndex[i - 1]], dr, dv,
405  velocity[chainIndex[i]], velocity[chainIndex[i - 1]],
406  velDepAcc[chainIndex[i]], velDepAcc[chainIndex[i - 1]]);
407  }
408 
409  for(int i = 2 ; i < size(); ++i)//update chain data(second near pair)
410  {
411  dr = chain.pos[i - 2] + chain.pos[i - 1];
412  dv = chainVel[i - 2] + chainVel[i - 1];
413  velDepForce(this->mass[chainIndex[i]], this->mass[chainIndex[i - 2]], dr, dv,
414  velocity[chainIndex[i]], velocity[chainIndex[i - 2]],
415  velDepAcc[chainIndex[i]], velDepAcc[chainIndex[i - 2]]);
416  }
417 
418  for(int i = 0 ; i < size(); ++i)//update non-chain data
419  {
420  for(int j = 0 ; j < i - 2; ++j)
421  {
422  dr = this->pos[chainIndex[i]] - this->pos[chainIndex[j]];
423  dv = velocity[chainIndex[i]] - velocity[chainIndex[j]];
424  velDepForce(this->mass[chainIndex[i]], this->mass[chainIndex[j]], dr, dv,
425  velocity[chainIndex[i]], velocity[chainIndex[j]],
426  velDepAcc[chainIndex[i]], velDepAcc[chainIndex[j]]);
427  }
428  }
429 
430  for(size_t i = 0 ; i < size() - 1; ++i)
431  chainVelDepAcc[i] = velDepAcc[chainIndex[i + 1]] - velDepAcc[chainIndex[i]];
432 
433  for(size_t i = 0 ; i < size() ; ++i)
434  this->acc[i] = chainVelIndepAcc[i] + chainVelDepAcc[i];
435 }
