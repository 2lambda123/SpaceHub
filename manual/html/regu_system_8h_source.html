296 template<typename Interaction, typename EvolvedData, typename Regularitor>
297 void reguSystem<Interaction, EvolvedData, Regularitor>::updateVelIndepAcc()
298 {
299  Vector dr(0.0, 0.0, 0.0);
300  Scalar inv_r = 1;
301  Scalar inv_r3 = 1;
302  memset(&(velIndepAcc[0]), 0, sizeof(Vector)*size());
303 
304  for(size_t i = 0 ; i < size() ; ++i)
305  {
306  for(size_t j = i + 1 ; j < size() ; ++j)
307  {
308  dr = this->pos[j] - this->pos[i];
309  inv_r = dr.reNorm();
310  inv_r3 = inv_r * inv_r * inv_r;
311  velIndepAcc[i] += dr * (inv_r3 * this->mass[j]);
312  velIndepAcc[j] -= dr * (inv_r3 * this->mass[i]);
313  }
314  }
315 }

322 template<typename Interaction, typename EvolvedData, typename Regularitor>
323 void reguSystem<Interaction, EvolvedData, Regularitor>::updateAccWith(VectorArray& velocity)
324 {
325  memset(&(velDepAcc[0]), 0, sizeof(Vector)*size());
326  Vector dr;
327  Vector dv;
328 
329  for(size_t i = 0 ; i < size() ; ++i)
330  {
331  for(size_t j = i + 1 ; j < size() ; ++j)
332  {
333  dr = this->pos[i] - this->pos[j];
334  dv = velocity[i] - velocity[j];
335  velDepForce(this->mass[i], this->mass[j], dr, dv,
336  velocity[i], velocity[j],
337  velDepAcc[i], velDepAcc[j]);
338  }
339  }
340 
341  for(size_t i = 0 ; i < size() ; ++i)
342  this->acc[i] = velIndepAcc[i] + velDepAcc[i];
343 }
